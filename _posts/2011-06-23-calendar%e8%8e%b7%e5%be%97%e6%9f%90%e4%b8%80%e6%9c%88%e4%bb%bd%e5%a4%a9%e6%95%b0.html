static int getLenOfMonth(Calendar Cal){ if(Cal.get(Cal.MONTH)>=0&&Cal.get(Cal.MONTH)<12) { Cal.set(Cal.DATE, 1); Cal.roll(Cal.DATE, -1); return (int)Cal.get(Cal.DATE); } else return 0; }
